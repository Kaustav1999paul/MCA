CREATE TABLE MUSICIAN (SSN VARCHAR(10), Name VARCHAR(30) NOT NULL, Address VARCHAR(50), Phone_No BIGINT, PRIMARY KEY(SSN));
CREATE TABLE INSTRUMENT(UID VARCHAR(10), Name VARCHAR(30) NOT NULL, Musical_Key VARCHAR(15), PRIMARY KEY(UID));
CREATE TABLE ALBUM(AID VARCHAR(10), Title VARCHAR(30), Copyright DATE, Format VARCHAR(15), SSN VARCHAR(10), PRIMARY KEY(AID), FOREIGN KEY(SSN) REFERENCES MUSICIAN(SSN));
CREATE TABLE SONG(SID VARCHAR(10), Title VARCHAR(30), AUTHOR VARCHAR(30), AID VARCHAR(10), PRIMARY KEY(SID), FOREIGN KEY(AID) REFERENCES ALBUM(AID));
CREATE TABLE MUS_INS(SSN VARCHAR(10), UID VARCHAR(10), PRIMARY KEY(SSN,UID), FOREIGN KEY(SSN) REFERENCES MUSICIAN(SSN), FOREIGN KEY(UID) REFERENCES INSTRUMENT(UID)); 
CREATE TABLE MUS_SONG(SSN VARCHAR(10), SID VARCHAR(10), PRIMARY KEY(SSN,SID), FOREIGN KEY(SSN) REFERENCES MUSICIAN(SSN), FOREIGN KEY(SID) REFERENCES SONG(SID));

SELECT M.Name AS MUSICIAN_NAME, S.Title AS SONG_TITLE, A.Title AS ALBUM_TITLE FROM MUSICIAN M, SONG S, ALBUM A, MUS_SONG MS WHERE M.SSN=MS.SSN AND MS.SID=S.SID AND S.AID=A.AID;

2.
SELECT M.SSN, M.Name FROM MUSICIAN M WHERE M.SSN NOT IN (SELECT A.SSN FROM ALBUM A, MUSICIAN M WHERE A.SSN=M.SSN);

3. SELECT S.Title as Song_Title, COUNT(MS.SID) AS Total_Musicians FROM MUS_SONG MS, SONG S WHERE S.SID=MS.SID GROUP BY(MS.SID) HAVING
